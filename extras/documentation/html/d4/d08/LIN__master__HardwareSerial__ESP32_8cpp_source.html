113{
114 // print debug message
115 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 2)
116 LIN_MASTER_DEBUG_SERIAL.println("LIN_Master_HardwareSerial_ESP32::_receiveFrame()");
117 #endif
118
119 // if state is wrong, exit immediately
120 if (this->state != LIN_Master_Base::STATE_BODY)
121 {
122 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_STATE);
123 this->state = LIN_Master_Base::STATE_DONE;
124 return this->state;
125 }
126
127 // optionally disable transmitter for slave response frames. Here, need to read BREAK as well due to delay of Serial.available()
128 if ((this->type == LIN_Master_Base::SLAVE_RESPONSE) && (this->pSerial->available() == 3))
129 _disableTransmitter();
130
131 // frame body received. Here, need to read BREAK as well due to delay of Serial.available()
132 if (this->pSerial->available() >= this->lenRx)
133 {
134 // store bytes in Rx
135 this->pSerial->readBytes(this->bufRx, this->lenRx);
136
137 // check frame for errors
138 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) this->_checkFrame());
139
140 // optionally disable transmitter after frame is completed
141 _disableTransmitter();
142
143 // progress state
144 this->state = LIN_Master_Base::STATE_DONE;
145
146 } // frame body received
147
148 // frame body received not yet received
149 else
150 {
151 // check for timeout
152 if (micros() - this->timeStart > this->timeMax)
153 {
154 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_TIMEOUT);
155 _disableTransmitter();
156 this->state = LIN_Master_Base::STATE_DONE;
157 }
158
159 } // not enough bytes received
160
161 // return state
162 return this->state;
