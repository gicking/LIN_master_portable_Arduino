77 return this->state;

79
80 // byte(s) received (likely BREAK echo)
81 if (this->pSerial->available())
82 {
83 // store echo in Rx
84 this->bufRx[0] = this->pSerial->read();
85
86 // restore nominal baudrate
87 this->pSerial->begin(this->baudrate);

90 // send rest of frame (request frame: SYNC+ID+DATA[]+CHK; response frame: SYNC+ID)
91 this->pSerial->write(this->bufTx+1, this->lenTx-1);
92
93 // progress state
94 this->state = LIN_Master_Base::STATE_BODY;
95
96 } // BREAK echo received
97
98 // no byte(s) received
99 else
100 {
101 // check for timeout
102 if (micros() - this->timeStart > this->timeMax)
103 {
104 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_TIMEOUT);
105 _disableTransmitter();
106 this->state = LIN_Master_Base::STATE_DONE;

124{
125 // print debug message (debug level 2)
126 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 2)
127 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
128 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_HardwareSerial::_receiveFrame()");
129 #endif
130
131 // if state is wrong, exit immediately
132 if (this->state != LIN_Master_Base::STATE_BODY)
133 {
134 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_STATE);
135 this->state = LIN_Master_Base::STATE_DONE;
136 return this->state;
137 }
138
139 // optionally disable transmitter for slave response frames. Len==2 because BREAK is handled already handled in _sendFrame()
140 if ((this->type == LIN_Master_Base::SLAVE_RESPONSE) && (this->pSerial->available() == 2))
141 _disableTransmitter();
142
143 // frame body received (-1 because BREAK is handled already handled in _sendFrame())
144 if (this->pSerial->available() >= this->lenRx-1)
145 {
146 // store bytes in Rx
147 this->pSerial->readBytes(this->bufRx+1, this->lenRx-1);
148
149 // check frame for errors
150 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) this->_checkFrame());
151
152 // optionally disable transmitter after frame is completed
153 _disableTransmitter();
154
155 // progress state
156 this->state = LIN_Master_Base::STATE_DONE;
157
158 } // frame body received
159
160 // frame body received not yet received
161 else
162 {
163 // check for timeout
164 if (micros() - this->timeStart > this->timeMax)
165 {
166 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_TIMEOUT);
167 _disableTransmitter();
168 this->state = LIN_Master_Base::STATE_DONE;
169 }
170
171 } // not enough bytes received
172
173 // return state
174 return this->state;

176} // LIN_Master_HardwareSerial::_receiveFrame()
