29uint8_t LIN_Master_Base::_calculatePID(void)
30{
31 uint8_t pid; // protected frame ID
32 uint8_t tmp; // temporary variable for calculating parity bits
33
34 // protect ID with parity bits
35 pid = (uint8_t) (this->id & 0x3F); // clear upper bits 6 & 7
36 tmp = (uint8_t) ((pid ^ (pid>>1) ^ (pid>>2) ^ (pid>>4)) & 0x01); // pid[6] = PI0 = ID0^ID1^ID2^ID4
37 pid |= (uint8_t) (tmp << 6);
38 tmp = (uint8_t) (~((pid>>1) ^ (pid>>3) ^ (pid>>4) ^ (pid>>5)) & 0x01); // pid[7] = PI1 = ~(ID1^ID3^ID4^ID5)
39 pid |= (uint8_t) (tmp << 7);
40
41 // print debug message (debug level 3)
42 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
43 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
44 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculatePID(): PID=0x");
45 LIN_MASTER_DEBUG_SERIAL.println((int) pid, HEX);
46 #endif
47
48 // return protected ID
49 return pid;
50
51} // LIN_Master_Base::_calculatePID()

62uint8_t LIN_Master_Base::_calculateChecksum(uint8_t NumData, uint8_t Data[])
63{
64 uint8_t pid; // protected frame ID
65 uint16_t chk=0x00;
66
67 // calculate protected frame ID
68 pid = this->_calculatePID();
69
70 // LIN2.x uses extended checksum which includes protected ID, i.e. including parity bits
71 // LIN1.x uses classical checksum only over data bytes
72 // Diagnostic frames with ID 0x3C and 0x3D/0x7D always use classical checksum (see LIN spec "2.3.1.5 Checkum")
73 if (!((this->version == LIN_V1) || (pid == 0x3C) || (pid == 0x7D))) // if version 2 & no diagnostic frames (0x3C=60 (PID=0x3C) or 0x3D=61 (PID=0x7D))
74 chk = (uint16_t) pid;
75
76 // loop over data bytes
77 for (uint8_t i = 0; i < NumData; i++)
78 {
79 chk += (uint16_t) (Data[i]);
80 if (chk>255)
81 chk -= 255;
82 }
83 chk = (uint8_t)(0xFF - ((uint8_t) chk)); // bitwise invert and strip upper byte
84
85 // print debug message (debug level 3)
86 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
87 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
88 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculateChecksum(): CHK=0x");
89 LIN_MASTER_DEBUG_SERIAL.println((int) chk, HEX);
90 #endif
91
92 // return frame checksum
93 return (uint8_t) chk;
94
95} // LIN_Master_Base::_calculateChecksum()

358LIN_Master_Base::state_t LIN_Master_Base::sendMasterRequest(LIN_Master_Base::version_t Version, uint8_t Id, uint8_t NumData, uint8_t Data[])
359{
360 // construct Tx frame
361 this->type = LIN_Master_Base::MASTER_REQUEST;
362 this->version = Version;
363 this->id = Id;
364 this->lenTx = NumData + 4; // Frame length
365 this->bufTx[0] = 0x00; // BREAK
366 this->bufTx[1] = 0x55; // SYNC
367 this->bufTx[2] = this->_calculatePID(); // PID
368 memcpy(this->bufTx+3, Data, NumData); // DATA[]
369 this->bufTx[this->lenTx-1] = _calculateChecksum(NumData, Data); // CHK
370 this->lenRx = this->lenTx; // just receive LIN echo
371
372 // init receive buffer
373 memset(this->bufRx, 0, 12);
374
375 // set break timeout (= 200% nominal) and start timeout
376 this->timeStart = micros();
377 this->timeoutFrame = ((this->lenRx + 1) * this->timePerByte) * 2;
378
379 // print debug message (debug level 2)
380 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 2)
381 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
382 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_Base::sendMasterRequest()");
383 #endif
384
385 // start LIN frame by sending a Sync Break
386 this->_sendBreak();
387
388 // return state machine state
389 return this->state;
390
391} // LIN_Master_Base::sendMasterRequest()

514LIN_Master_Base::state_t LIN_Master_Base::handler(void)
515{
516 // print debug message (debug level 3)
517 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
518 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
519 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::handler(: state = ");
520 LIN_MASTER_DEBUG_SERIAL.println((int) this->state);
521 #endif
522
523 // act according to current state
524 switch (this->state)
525 {
526 // idle or done -> don't do anything
527 case LIN_Master_Base::STATE_IDLE:
528 case LIN_Master_Base::STATE_DONE:
529 break;
530
531 // when sync break done, send rest of frame
532 case LIN_Master_Base::STATE_BREAK:
533 this->_sendFrame();
534 break;
535
536 // when frame done, read and check it
537 case LIN_Master_Base::STATE_BODY:
538 this->_receiveFrame();
539 break;
540
541 // this should never happen..
542 default:
543
544 // print debug message (debug level 1)
545 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 1)
546 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
547 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::handler(): ");
548 LIN_MASTER_DEBUG_SERIAL.print("wrong state 0x");
549 LIN_MASTER_DEBUG_SERIAL.println(this->state, HEX);
550 #endif
551
552 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_MISC);
553 this->state = LIN_Master_Base::STATE_DONE;
554 _disableTransmitter();
555
556 } // switch (state)
557
558 // return state machine state
559 return this->state;
560
561} // LIN_Master_Base::handler()
