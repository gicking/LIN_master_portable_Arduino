22uint8_t LIN_Master_Base::_calculatePID(void)
23{
24 uint8_t pid; // protected frame ID
25 uint8_t tmp; // temporary variable for calculating parity bits
26
27 // protect ID with parity bits
28 pid = (uint8_t) (this->id & 0x3F); // clear upper bits 6 & 7
29 tmp = (uint8_t) ((pid ^ (pid>>1) ^ (pid>>2) ^ (pid>>4)) & 0x01); // pid[6] = PI0 = ID0^ID1^ID2^ID4
30 pid |= (uint8_t) (tmp << 6);
31 tmp = (uint8_t) (~((pid>>1) ^ (pid>>3) ^ (pid>>4) ^ (pid>>5)) & 0x01); // pid[7] = PI1 = ~(ID1^ID3^ID4^ID5)
32 pid |= (uint8_t) (tmp << 7);
33
34 // print debug message (debug level 3)
35 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
36 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
37 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculatePID(): PID=0x");
38 LIN_MASTER_DEBUG_SERIAL.println((int) pid, HEX);
39 #endif
40
41 // return protected ID
42 return pid;
43
44} // LIN_Master_Base::_calculatePID()

55uint8_t LIN_Master_Base::_calculateChecksum(uint8_t NumData, uint8_t Data[])
56{
57 uint8_t pid; // protected frame ID
58 uint16_t chk=0x00;
59
60 // calculate protected frame ID
61 pid = this->_calculatePID();
62
63 // LIN2.x uses extended checksum which includes protected ID, i.e. including parity bits
64 // LIN1.x uses classical checksum only over data bytes
65 // Diagnostic frames with ID 0x3C and 0x3D/0x7D always use classical checksum (see LIN spec "2.3.1.5 Checkum")
66 if (!((this->version == LIN_V1) || (pid == 0x3C) || (pid == 0x7D))) // if version 2 & no diagnostic frames (0x3C=60 (PID=0x3C) or 0x3D=61 (PID=0x7D))
67 chk = (uint16_t) pid;
68
69 // loop over data bytes
70 for (uint8_t i = 0; i < NumData; i++)
71 {
72 chk += (uint16_t) (Data[i]);
73 if (chk>255)
74 chk -= 255;
75 }
76 chk = (uint8_t)(0xFF - ((uint8_t) chk)); // bitwise invert and strip upper byte
77
78 // print debug message (debug level 3)
79 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
80 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
81 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculateChecksum(): CHK=0x");
82 LIN_MASTER_DEBUG_SERIAL.println((int) chk, HEX);
83 #endif
84
85 // return frame checksum
86 return (uint8_t) chk;
87
88} // LIN_Master_Base::_calculateChecksum()

97LIN_Master_Base::error_t LIN_Master_Base::_checkFrame(void)
98{
99 // print debug message (debug level 3)
100 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
101 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
102 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_Base::_checkFrame()");
103 #endif
104
105 // check echo of sent bytes (frame or header). Exit on mismatch
106 for (uint8_t i=0; i<this->lenTx; i++)
107 if (this->bufTx[i] != this->bufRx[i])
108 return LIN_Master_Base::ERROR_ECHO;
109
110 // check frame checksum
111 if (this->bufRx[lenRx-1] != _calculateChecksum(this->lenRx-4, this->bufRx+3))
112 return LIN_Master_Base::ERROR_CHK;
113
114 // return result of check
115 return LIN_Master_Base::NO_ERROR;
116
117} // LIN_Master_Base::_checkFrame()

295LIN_Master_Base::state_t LIN_Master_Base::sendMasterRequest(LIN_Master_Base::version_t Version, uint8_t Id, uint8_t NumData, uint8_t Data[])
296{
297 // construct Tx frame
298 this->type = LIN_Master_Base::MASTER_REQUEST;
299 this->version = Version;
300 this->id = Id;
301 this->lenTx = NumData + 4; // Frame length
302 this->bufTx[0] = 0x00; // BREAK
303 this->bufTx[1] = 0x55; // SYNC
304 this->bufTx[2] = this->_calculatePID(); // PID
305 memcpy(this->bufTx+3, Data, NumData); // DATA[]
306 this->bufTx[this->lenTx-1] = _calculateChecksum(NumData, Data); // CHK
307 this->lenRx = this->lenTx; // just receive LIN echo
308
309 // init receive buffer
310 memset(this->bufRx, 0, 12);
311
312 // set break timeout (= 150% nominal) and start timeout
313 this->timeStart = micros();
314 this->timeMax = (((this->lenRx + 1) * this->timePerByte) * 3 ) >> 1;
315
316 // print debug message (debug level 2)
317 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 2)
318 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
319 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_Base::sendMasterRequest()");
320 #endif
321
322 // start LIN frame by sending a Sync Break
323 this->_sendBreak();
324
325 // return state machine state
326 return this->state;
327
328} // LIN_Master_Base::sendMasterRequest()

451LIN_Master_Base::state_t LIN_Master_Base::handler(void)
452{
453 // print debug message (debug level 3)
454 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
455 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
456 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::handler(: state = ");
457 LIN_MASTER_DEBUG_SERIAL.println((int) this->state);
458 #endif
459
460 // act according to current state
461 switch (this->state)
462 {
463 // idle or done -> don't do anything
464 case LIN_Master_Base::STATE_IDLE:
465 case LIN_Master_Base::STATE_DONE:
466 break;
467
468 // when sync break done, send rest of frame
469 case LIN_Master_Base::STATE_BREAK:
470 this->_sendFrame();
471 break;
472
473 // when frame done, read and check it
474 case LIN_Master_Base::STATE_BODY:
475 this->_receiveFrame();
476 break;
477
478 // this should never happen..
479 default:
480 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_MISC);
481 _disableTransmitter();
482 this->state = LIN_Master_Base::STATE_DONE;
483
484 } // switch (state)
485
486 // return state machine state
487 return this->state;
488
489} // LIN_Master_Base::handler()
