28uint8_t LIN_Master_Base::_calculatePID(void)
29{
30 uint8_t pid; // protected frame ID
31 uint8_t tmp; // temporary variable for calculating parity bits
32
33 // protect ID with parity bits
34 pid = (uint8_t) (this->id & 0x3F); // clear upper bits 6 & 7
35 tmp = (uint8_t) ((pid ^ (pid>>1) ^ (pid>>2) ^ (pid>>4)) & 0x01); // pid[6] = PI0 = ID0^ID1^ID2^ID4
36 pid |= (uint8_t) (tmp << 6);
37 tmp = (uint8_t) (~((pid>>1) ^ (pid>>3) ^ (pid>>4) ^ (pid>>5)) & 0x01); // pid[7] = PI1 = ~(ID1^ID3^ID4^ID5)
38 pid |= (uint8_t) (tmp << 7);
39
40 // print debug message (debug level 3)
41 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
42 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
43 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculatePID(): PID=0x");
44 LIN_MASTER_DEBUG_SERIAL.println((int) pid, HEX);
45 #endif
46
47 // return protected ID
48 return pid;
49
50} // LIN_Master_Base::_calculatePID()

61uint8_t LIN_Master_Base::_calculateChecksum(uint8_t NumData, uint8_t Data[])
62{
63 uint8_t pid; // protected frame ID
64 uint16_t chk=0x00;
65
66 // calculate protected frame ID
67 pid = this->_calculatePID();
68
69 // LIN2.x uses extended checksum which includes protected ID, i.e. including parity bits
70 // LIN1.x uses classical checksum only over data bytes
71 // Diagnostic frames with ID 0x3C and 0x3D/0x7D always use classical checksum (see LIN spec "2.3.1.5 Checkum")
72 if (!((this->version == LIN_V1) || (pid == 0x3C) || (pid == 0x7D))) // if version 2 & no diagnostic frames (0x3C=60 (PID=0x3C) or 0x3D=61 (PID=0x7D))
73 chk = (uint16_t) pid;
74
75 // loop over data bytes
76 for (uint8_t i = 0; i < NumData; i++)
77 {
78 chk += (uint16_t) (Data[i]);
79 if (chk>255)
80 chk -= 255;
81 }
82 chk = (uint8_t)(0xFF - ((uint8_t) chk)); // bitwise invert and strip upper byte
83
84 // print debug message (debug level 3)
85 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
86 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
87 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::_calculateChecksum(): CHK=0x");
88 LIN_MASTER_DEBUG_SERIAL.println((int) chk, HEX);
89 #endif
90
91 // return frame checksum
92 return (uint8_t) chk;
93
94} // LIN_Master_Base::_calculateChecksum()

103LIN_Master_Base::error_t LIN_Master_Base::_checkFrame(void)
104{
105 // print debug message (debug level 3)
106 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
107 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
108 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_Base::_checkFrame()");
109 #endif
110
111 // check echo of sent bytes (frame or header). Exit on mismatch
112 for (uint8_t i=0; i<this->lenTx; i++)
113 if (this->bufTx[i] != this->bufRx[i])
114 return LIN_Master_Base::ERROR_ECHO;
115
116 // check frame checksum
117 if (this->bufRx[lenRx-1] != _calculateChecksum(this->lenRx-4, this->bufRx+3))
118 return LIN_Master_Base::ERROR_CHK;
119
120 // return result of check
121 return LIN_Master_Base::NO_ERROR;

301LIN_Master_Base::state_t LIN_Master_Base::sendMasterRequest(LIN_Master_Base::version_t Version, uint8_t Id, uint8_t NumData, uint8_t Data[])
302{
303 // construct Tx frame
304 this->type = LIN_Master_Base::MASTER_REQUEST;
305 this->version = Version;
306 this->id = Id;
307 this->lenTx = NumData + 4; // Frame length
308 this->bufTx[0] = 0x00; // BREAK
309 this->bufTx[1] = 0x55; // SYNC
310 this->bufTx[2] = this->_calculatePID(); // PID
311 memcpy(this->bufTx+3, Data, NumData); // DATA[]
312 this->bufTx[this->lenTx-1] = _calculateChecksum(NumData, Data); // CHK
313 this->lenRx = this->lenTx; // just receive LIN echo
314
315 // init receive buffer
316 memset(this->bufRx, 0, 12);
317
318 // set break timeout (= 150% nominal) and start timeout
319 this->timeStart = micros();
320 this->timeMax = (((this->lenRx + 1) * this->timePerByte) * 3 ) >> 1;
321
322 // print debug message (debug level 2)
323 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 2)
324 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
325 LIN_MASTER_DEBUG_SERIAL.println(": LIN_Master_Base::sendMasterRequest()");
326 #endif
327
328 // start LIN frame by sending a Sync Break
329 this->_sendBreak();
330
331 // return state machine state
332 return this->state;
333
334} // LIN_Master_Base::sendMasterRequest()

457LIN_Master_Base::state_t LIN_Master_Base::handler(void)
458{
459 // print debug message (debug level 3)
460 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
461 LIN_MASTER_DEBUG_SERIAL.print(this->nameLIN);
462 LIN_MASTER_DEBUG_SERIAL.print(": LIN_Master_Base::handler(: state = ");
463 LIN_MASTER_DEBUG_SERIAL.println((int) this->state);
464 #endif
465
466 // act according to current state
467 switch (this->state)
468 {
469 // idle or done -> don't do anything
470 case LIN_Master_Base::STATE_IDLE:
471 case LIN_Master_Base::STATE_DONE:
472 break;
473
474 // when sync break done, send rest of frame
475 case LIN_Master_Base::STATE_BREAK:
476 this->_sendFrame();
477 break;
478
479 // when frame done, read and check it
480 case LIN_Master_Base::STATE_BODY:
481 this->_receiveFrame();
482 break;
483
484 // this should never happen..
485 default:
486 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_MISC);
487 _disableTransmitter();
488 this->state = LIN_Master_Base::STATE_DONE;
489
490 } // switch (state)
491
492 // return state machine state
493 return this->state;
494
495} // LIN_Master_Base::handler()
