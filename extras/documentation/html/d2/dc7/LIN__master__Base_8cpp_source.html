22uint8_t LIN_Master_Base::_calculatePID(void)
23{
24 uint8_t pid; // protected frame ID
25 uint8_t tmp; // temporary variable for calculating parity bits
26
27 // copy (unprotected) ID
28 pid = this->id;
29
30 // protect ID with parity bits
31 pid = (uint8_t) (pid & 0x3F); // clear upper bits 6 & 7
32 tmp = (uint8_t) ((pid ^ (pid>>1) ^ (pid>>2) ^ (pid>>4)) & 0x01); // pid[6] = PI0 = ID0^ID1^ID2^ID4
33 pid |= (uint8_t) (tmp << 6);
34 tmp = (uint8_t) (~((pid>>1) ^ (pid>>3) ^ (pid>>4) ^ (pid>>5)) & 0x01); // pid[7] = PI1 = ~(ID1^ID3^ID4^ID5)
35 pid |= (uint8_t) (tmp << 7);
36
37 // return protected ID
38 return pid;
39
40} // LIN_Master_Base::_calculatePID()

51uint8_t LIN_Master_Base::_calculateChecksum(uint8_t NumData, uint8_t Data[])
52{
53 uint8_t pid; // protected frame ID
54 uint16_t chk=0x00;
55
56 // calculate protected frame ID
57 pid = this->_calculatePID();
58
59 // LIN2.x uses extended checksum which includes protected ID, i.e. including parity bits
60 // LIN1.x uses classical checksum only over data bytes
61 // Diagnostic frames with ID 0x3C and 0x3D/0x7D always use classical checksum (see LIN spec "2.3.1.5 Checkum")
62 if (!((this->version == LIN_V1) || (pid == 0x3C) || (pid == 0x7D))) // if version 2 & no diagnostic frames (0x3C=60 (PID=0x3C) or 0x3D=61 (PID=0x7D))
63 chk = (uint16_t) pid;
64
65 // loop over data bytes
66 for (uint8_t i = 0; i < NumData; i++)
67 {
68 chk += (uint16_t) (Data[i]);
69 if (chk>255)
70 chk -= 255;
71 }
72 chk = (uint8_t)(0xFF - ((uint8_t) chk)); // bitwise invert
73
74 // return frame checksum
75 return (uint8_t) chk;
76
77} // LIN_Master_Base::_calculateChecksum()

274LIN_Master_Base::state_t LIN_Master_Base::sendMasterRequest(LIN_Master_Base::version_t Version, uint8_t Id, uint8_t NumData, uint8_t Data[])
275{
276 // construct Tx frame
277 this->type = LIN_Master_Base::MASTER_REQUEST;
278 this->version = Version;
279 this->id = Id;
280 this->lenTx = NumData + 4; // Frame length
281 this->bufTx[0] = 0x00; // BREAK
282 this->bufTx[1] = 0x55; // SYNC
283 this->bufTx[2] = this->_calculatePID(); // PID
284 memcpy(this->bufTx+3, Data, NumData); // DATA[]
285 this->bufTx[this->lenTx-1] = _calculateChecksum(NumData, Data); // CHK
286 this->lenRx = this->lenTx; // just receive LIN echo
287
288 // init receive buffer
289 memset(this->bufRx, 0, 12);
290
291 // set break timeout (= 150% nominal) and start timeout
292 this->timeStart = micros();
293 this->timeMax = (((this->lenRx + 1) * this->timePerByte) * 3 ) >> 1;
294
295 // start LIN frame by sending a Sync Break
296 this->_sendBreak();
297
298 // return state machine state
299 return this->state;
300
301} // LIN_Master_Base::sendMasterRequest()

406LIN_Master_Base::state_t LIN_Master_Base::handler(void)
407{
408 // act according to current state
409 switch (this->state)
410 {
411 // idle or done -> don't do anything
412 case LIN_Master_Base::STATE_IDLE:
413 case LIN_Master_Base::STATE_DONE:
414 break;
415
416 // when sync break done, send rest of frame
417 case LIN_Master_Base::STATE_BREAK:
418 this->_sendFrame();
419 break;
420
421 // when frame done, read and check it
422 case LIN_Master_Base::STATE_BODY:
423 this->_receiveFrame();
424 break;
425
426 // this should never happen..
427 default:
428 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_MISC);
429 _disableTransmitter();
430 this->state = LIN_Master_Base::STATE_DONE;
431
432 } // switch (state)
433
434 // return state machine state
435 return this->state;
436
437} // LIN_Master_Base::handler()
