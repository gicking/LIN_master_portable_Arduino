70{
71 // if state is wrong, exit immediately
72 if (this->state != LIN_Master_Base::STATE_BREAK)

74 // print debug message

80 this->_disableTransmitter();
81 return this->state;
82 }
83
84 // byte(s) received (likely BREAK echo)
85 if (this->pSerial->available())
86 {
87 // store echo in Rx
88 this->bufRx[0] = this->pSerial->read();
89
90 // no baudrate update due to long delay, see https://github.com/stm32duino/Arduino_Core_STM32/issues/2907
91 // Instead BREAK is generated using dedicated LIN mode above
92
93 // send rest of frame (request frame: SYNC+ID+DATA[]+CHK; response frame: SYNC+ID)
94 this->pSerial->write(this->bufTx+1, this->lenTx-1);
95
96 // progress state
97 this->state = LIN_Master_Base::STATE_BODY;
98
99 } // BREAK echo received
100
101 // no byte(s) received
102 else
103 {
104 // check for timeout
105 if (micros() - this->timeStart > this->timeoutFrame)
106 {
107 // print debug message
108 DEBUG_PRINT(1, "Rx timeout");
109
110 // set error state and return immediately
111 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_TIMEOUT);
112 this->state = LIN_Master_Base::STATE_DONE;
113 this->_disableTransmitter();
114 return this->state;
115 }

117 } // no byte(s) received
118
119 // print debug message

134LIN_Master_Base::state_t LIN_Master_HardwareSerial_STM32::_receiveFrame(void)
135{
136 // if state is wrong, exit immediately
137 if (this->state != LIN_Master_Base::STATE_BODY)
138 {
139 // print debug message
140 DEBUG_PRINT(1, "wrong state 0x%02X", this->state);
141
142 // set error state and return immediately
143 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_STATE);
144 this->state = LIN_Master_Base::STATE_DONE;
145 this->_disableTransmitter();
146 return this->state;
147 }
148
149 // optionally disable RS485 transmitter for slave response frames. Len==2 because BREAK is handled already handled in _sendFrame()
150 if ((this->type == LIN_Master_Base::SLAVE_RESPONSE) && (this->pSerial->available() == 2))
151 this->_disableTransmitter();
152
153 // frame body received (-1 because BREAK is handled already handled in _sendFrame())
154 if (this->pSerial->available() >= this->lenRx-1)
155 {
156 // store bytes in Rx
157 this->pSerial->readBytes(this->bufRx+1, this->lenRx-1);
158
159 // check frame for errors
160 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) this->_checkFrame());
161
162 // optionally disable RS485 transmitter after frame is completed
163 this->_disableTransmitter();
164
165 // progress state
166 this->state = LIN_Master_Base::STATE_DONE;
167
168 } // frame body received
169
170 // frame body received not yet received
171 else
172 {
173 // check for timeout
174 if (micros() - this->timeStart > this->timeoutFrame)
175 {
176 // print debug message
177 DEBUG_PRINT(1, "Rx timeout");
178
179 // set error state and return immediately
180 this->error = (LIN_Master_Base::error_t) ((int) this->error | (int) LIN_Master_Base::ERROR_TIMEOUT);
181 this->state = LIN_Master_Base::STATE_DONE;
182 this->_disableTransmitter();
183 return this->state;
184 }
185
186 } // not enough bytes received
187
188 // print debug message
189 DEBUG_PRINT(2, " ");
190
191 // return state
192 return this->state;

227void LIN_Master_HardwareSerial_STM32::begin(uint16_t Baudrate)
228{
229 // call base class method
230 LIN_Master_Base::begin(Baudrate);
231
232 // just to be sure
233 this->pSerial->end();
234
235 // remap Rx/Tx pins (before begin()!)
236 this->pSerial->setTx(this->pinTx);
237 this->pSerial->setRx(this->pinRx);
238
239 // open serial interface with optional timeout
240 this->pSerial->begin(this->baudrate);
241 #if defined(LIN_MASTER_LIN_PORT_TIMEOUT) && (LIN_MASTER_LIN_PORT_TIMEOUT > 0)
242 uint32_t startMillis = millis();
243 while ((!(*(this->pSerial))) && (millis() - startMillis < LIN_MASTER_LIN_PORT_TIMEOUT));
244 #else
245 while(!(*(this->pSerial)));
246 #endif
247
248 // enable LIN mode -> BREAK 13b & enable BREAK detection
249 this->huart->Instance->CR1 &= ~USART_CR1_UE;
250 this->huart->Instance->CR2 |= USART_CR2_LINEN;
251 this->huart->Instance->CR1 |= USART_CR1_UE;
252
253 // print debug message
254 DEBUG_PRINT(2, "ok");
255
256} // LIN_Master_HardwareSerial_STM32::begin()
