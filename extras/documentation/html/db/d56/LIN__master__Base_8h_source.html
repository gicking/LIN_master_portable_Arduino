301 inline void getFrame(LIN_Master_Base::frame_t &Type, uint8_t &Id, uint8_t &NumData, uint8_t Data[])
302 {
303 // print debug message (debug level 3)
304 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)

306 #endif
307
308 noInterrupts(); // for data consistency temporarily disable ISRs
309 Type = this->type; // frame type
310 Id = this->id; // frame ID
311 NumData = this->lenRx - 4; // number of data bytes (excl. BREAK, SYNC, ID, CHK)
312 memcpy(Data, this->bufRx+3, NumData); // copy data bytes
313 interrupts(); // re-enable ISRs
314
315 } // getFrame()
