243 inline void getFrame(LIN_Master_Base::frame_t &Type, uint8_t &Id, uint8_t &NumData, uint8_t Data[])
244 {
245 // print debug message (debug level 3)
246 #if defined(LIN_MASTER_DEBUG_SERIAL) && (LIN_MASTER_DEBUG_LEVEL >= 3)
247 LIN_MASTER_DEBUG_SERIAL.println("LIN_Master_Base::getFrame()");
248 #endif
249
250 noInterrupts(); // for data consistency temporarily disable ISRs
251 Type = this->type; // frame type
252 Id = this->id; // frame ID
253 NumData = this->lenRx - 4; // number of data bytes (excl. BREAK, SYNC, ID, CHK)
254 memcpy(Data, this->bufRx+3, NumData); // copy data bytes
255 interrupts(); // re-enable ISRs
256
257 } // getFrame()
